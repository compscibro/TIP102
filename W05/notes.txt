# Multiple Pass Technique
'''
phases:
    from the end
    middle of the LinkedList
    based on the size of the LinkedList
'''
def m_pass(head):

    # Pass 1: compute global info

        # collect info (length, sum, etc.)
        curr = curr.next
    curr = head
    # process using that info
    while curr:
        # perform the operation using computed info on first pass
        curr = curr.next

# Slow-Fast Pointer Technique
'''
Use two pointers moving at different speeds and or time to solve problems involving cycles
or middle elements. 
Common applications:
- Detecting a cycle in a LinkedList
- Finding the middle
'''
def slow_fast(head):
    slow = head
    fast = head
    while fast and fast.next:
        slow = slow.next
        fast = fast.next.next
    return slow # middle node
